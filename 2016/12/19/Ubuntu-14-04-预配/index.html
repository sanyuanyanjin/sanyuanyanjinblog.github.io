Ubuntu 14.04 预配

apt-get install vim
apt-get install openssh-client=1:6.6p1-2ubuntu1
apt-get install openssh-server
vim /etc/ssh/sshd_config
找到PermitRootLogin no一行，改为PermitRootLogin yes
PermitRootLogin yes

PermitRootLogin without-password

root用户登录桌面
vim /usr/share/lightdm/lightdm.conf.d/50-ubuntu.conf
[SeatDefaults]
user-session=ubuntu
greeter-show-manual-login=true

apt-get install lrzsz

root@ubuntu:~# mkdir /usr/lib/java
root@ubuntu:~# mv jdk1.8.0_91/ /usr/lib/java
root@ubuntu:~# cd /usr/lib/java/jdk1.8.0_91/
root@ubuntu:/usr/lib/java/jdk1.8.0_91#
chmod a+x -R /usr/lib/java/jdk1.8.0_91/

vim ~/.bashrc

vim /etc/profile
export JAVA_HOME=/usr/lib/java/jdk1.8.0_91/
export JRE_HOME=${JAVA_HOME}/jre
export CLASS_PATH=.:${JAVA_HOME}/lib:${JRE_HOME}/lib
export PATH=${JAVA_HOME}/bin:$PATH

root@jinyan-virtual-machine:~# source /etc/profile
