#!/bin/bash
host=”10.6.100.39”
id=”ctzapc”
pw=”htw001c9”
basedir=”/home/“
remotedir=”/backup_dhcp”
tar -cjPf $basedir/dhcp.$(date +%Y-%m-%d).tar.bz2 /home/backup/
backupfile=dhcp.$(date +%Y-%m-%d).tar.bz2
rmfile=dhcp.$(date +%Y-%m-%d –date -7day).tar.bz2
ftp -n “$host” <<EOF
user $id $pw
binary
cd $remotedir
lcd $basedir
put $backupfile
bye
EOF
rm -f $basedir/$rmfile
